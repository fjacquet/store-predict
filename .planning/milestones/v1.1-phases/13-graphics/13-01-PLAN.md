    - DARK_GREY_RL = colors.HexColor("#ADB5BD")
    - LIGHT_GREY_RL = colors.HexColor("#CED4DA")

    **make_drr_bar_drawing(summary, width=400, height=180) -> Drawing:**
    - Guard: if no workload_groups, return Drawing(width, height).
    - bc = VerticalBarChart(); set x=40, y=20, width=width-60, height=height-40.
    - categories = [grp.category[:20] for grp in summary.workload_groups] (truncate for PDF label fit).
    - bc.data = [tuple(round(grp.avg_drr, 2) for grp in summary.workload_groups)] — list of one tuple (single series).
    - bc.categoryAxis.categoryNames = categories, labels.angle=30, labels.boxAnchor="ne".
    - bc.bars[0].fillColor = DELL_BLUE_RL.
    - bc.valueAxis.valueMin = 0.
    - d.add(bc); return d.

    **make_before_after_bar_drawing(summary, width=500, height=160) -> Drawing:**
    - Guard: if no workload_groups, return Drawing(width, height).
    - bc = VerticalBarChart(); x=50, y=20, width=width-70, height=height-40.
    - provisioned_tuple = tuple(round(grp.total_provisioned_mib / 1024, 1) for grp in summary.workload_groups)
    - required_tuple = tuple(round(grp.total_required_mib / 1024, 1) for grp in summary.workload_groups)
    - bc.data = [provisioned_tuple, required_tuple] (two series = grouped bars).
    - categories = [grp.category[:20] for grp in summary.workload_groups].
    - bc.categoryAxis.categoryNames = categories, labels.angle=30, labels.boxAnchor="ne".
    - bc.bars[0].fillColor = DELL_BLUE_RL (Provisioned).
    - bc.bars[1].fillColor = LIGHT_BLUE_RL (Required).
    - bc.valueAxis.valueMin = 0.
    - d.add(bc); return d.

    **make_pie_drawing(summary, width=250, height=200) -> Drawing:**
    - Guard: if no workload_groups, return Drawing(width, height).
    - pc = Pie(); pc.x=60, pc.y=20, pc.width=130, pc.height=130.
    - pc.data = [grp.total_provisioned_mib for grp in summary.workload_groups].
    - pc.labels = [grp.category[:15] for grp in summary.workload_groups] (truncate for pie label fit).
    - Apply DELL_PALETTE_RL colors: `for i in range(len(pc.data)): pc.slices[i].fillColor = DELL_PALETTE_RL[i % len(DELL_PALETTE_RL)]`.
    - d.add(pc); return d.

    **make_sankey_image_flowable(summary, width_pt=500, height_pt=200) -> Image:**
    - Lazy imports inside function body:
      ```python
      import matplotlib
      matplotlib.use("Agg")  # Must be before pyplot import
      import matplotlib.pyplot as plt
      from matplotlib.sankey import Sankey
      ```
    - Guard: if not summary.workload_groups or summary.total_provisioned_mib == 0, return empty Image(BytesIO(b""), width=width_pt, height=height_pt).
    - Build flows: [total_provisioned] + [-grp.total_required_mib for grp in summary.workload_groups].
    - Build labels: ["Provisioned"] + [f"{grp.category[:12]}\n{grp.avg_drr:.1f}x" for grp in summary.workload_groups].
    - orientations: [0] + [-1 for _ in summary.workload_groups].
    - fig, ax = plt.subplots(figsize=(width_pt/72, height_pt/72), dpi=150).
    - ax.set_axis_off(); ax.set_xticks([]); ax.set_yticks([]).
    - scale = 1.0 / summary.total_provisioned_mib.
    - try: sankey = Sankey(ax=ax, scale=scale, offset=0.15, unit="GiB", format="%.0f"); sankey.add(flows=flows, labels=labels, orientations=orientations, pathlengths=[0.2]*len(flows), patchlabel="Data\nReduction", facecolor="#007DB8"); sankey.finish(); buf = BytesIO(); fig.savefig(buf, format="png", dpi=150, bbox_inches="tight", facecolor="white", transparent=False); buf.seek(0); return Image(buf, width=width_pt, height=height_pt)
    - finally: plt.close(fig) — prevent memory leak.

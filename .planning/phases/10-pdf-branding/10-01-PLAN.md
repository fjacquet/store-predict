Helper to generate test PNG bytes:
```python
def _make_png_bytes(mode: str = "RGBA", size: tuple[int, int] = (100, 50)) -> bytes:
    from PIL import Image
    from io import BytesIO
    img = Image.new(mode, size, color=(255, 0, 0, 128) if "A" in mode else (255, 0, 0))
    if mode == "P":
        img = img.convert("P")
    buf = BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()

def _make_jpeg_bytes(size: tuple[int, int] = (100, 50)) -> bytes:
    from PIL import Image
    from io import BytesIO
    img = Image.new("RGB", size, color=(255, 0, 0))
    buf = BytesIO()
    img.save(buf, format="JPEG")
    return buf.getvalue()
```
